variance = sum (Y .* Y, 1) / n
ecart = sqrt (variance)
Dinvs = diag (ones(ecart)./ecart)
Z = Y * Dinvs
R = Dinvs * V * Dinvs
R1 = Z' * D * Z
R-R1
Dinvs2 = diag (ones(variance)./variance)
M = Dinvs2
MV = M * V
